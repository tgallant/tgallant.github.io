'use strict';

var e1 = function(x) {
  return function(a,b) {
    if(a % 3 == 0 || a % 5 == 0) {
      return a < x ? arguments.callee(a+1,b+a) : b;
    }
    else {
      return arguments.callee(a+1,b);
    }
  }(1,0);
};
