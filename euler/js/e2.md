    'use strict';

    var fib = function(n) {
      return function(n,a,b) {
        return n>0 ? arguments.callee(n-1,b,a+b) : a;
      }(n,0,1);
    };

    var e2 = function(x) {
      return function(a,b) {
        var fibNum = fib(a);
        if(fibNum % 2 == 0) {
          return fibNum < x ? arguments.callee(a+1,fibNum+b) : b;
        }
        else {
          return arguments.callee(a+1,b);
        }
      }(2,0);
    };
