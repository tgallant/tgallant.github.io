    'use strict';

    var e3 = function(x) {
      return function(a,b) {
        if(a > 1) { 
          if(a % b == 0) {
            return arguments.callee(a/b,b-1);
          }
          else {
            return arguments.callee(a,b+1);
          }
        }
        else {
          return b+1;
        }
      }(x,2);
    };
