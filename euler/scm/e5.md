<pre><code>
;; helpers
 
(define (sum l)
  (if (null? l)
      0
      (+ (car l) (sum (cdr l)))))
 
(define (sqr x)
  (* x x))
 
;; e5
 
(define (sum-of-sq x)
  (sum (map sqr (cdr (iota x)))))
 
(define (sq-of-sum x)
  (sqr (sum (cdr (iota x)))))
 
(define (e6 x)
  (- (sq-of-sum x) (sum-of-sq x)))
